00591 //              const QVMatrix kkT = k.toColumnMatrix() * k.toRowMatrix(),
00592 //              const QVMatrix k_x = k.crossProductMatrix();
00593 //      Rotaciones:
00594 //              const QVMatrix R1 = cos(theta)*I+ sin(theta)*k_x + (1-cos(theta))*kkT;
00595 //              const QVMatrix R2 = I + sin(theta)*k_x + (1-cos(theta))* k_x*k_x;
00596 //              const QVMatrix R3 = I + sin(theta)*k_x + (1-cos(theta))*(kkT-I);
00597 //
00598 QVMatrix getRotation(const double r1, const double r2, const double r3, const double Epsi)
00599     {
00600     const double t1 = r1 * r1;
00601     const double t2 = r2 * r2;
00602     const double t3 = r3 * r3;
00603     const double t5 = sqrt(t1 + t2 + t3);
00604     const double t6 = t5 + Epsi;
00605     const double t7 = cos(t6);
00606     const double t9 = t6 * t6;
00607     const double t11 = (0.1e1 - t7) / t9;
00608     const double t14 = sin(t6);
00609     const double t16 = t14 / t6;
00610     const double t17 = t16 * r3;
00611     const double t19 = t11 * r1 * r2;
00612     const double t21 = t16 * r2;
00613     const double t23 = t11 * r1 * r3;

00616

00618     cg[0][0] = t7 + t11 * t1;
00619     cg[0][1] = -t17 + t19;
00620     cg[0][2] = t21 + t23;
00621     cg[1][0] = t17 + t19;
00622     cg[1][1] = t7 + t11 * t2;
00623     cg[1][2] = -t28 + t30;

00626     cg[2][2] = t7 + t11 * t3;
